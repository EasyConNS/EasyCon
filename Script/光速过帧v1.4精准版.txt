# <<光速过帧>>
# ver 1.4 (稳定版),可以自定义延迟,支持联机
# 实测每小时约6200帧，必须使用v4或更高版本固件
# 联机速度稍慢5700左右，而且会受cpu100%影响
# 使用方法：
# 1.站在室内，用对战进入可改日期状态，存个档以防万一
# 2.进系统菜单，关掉时间同步
# 3.打开日期修改，选择一个31天的月份，日期设为1号
# 4.光标移到最右边后退出，确保再打开时光标在OK按钮上
# 5.烧录脚本后运行或者固件模式也行

# 这里帧数你是实际要过的帧数,要过多少填多少
_帧数 = 10000

# 这里是延迟，如果默认30的情况下依然跑飞，建议每次增加10%左右，也就是增加3，然后尝试
_延迟 = 30
# 如果联机将下面的90改为120
$4 = 120



$3 = _延迟
$4 += _延迟
# 检查连接
FOR 10
    RCLICK
    300
NEXT
# 过帧
$1 = 1
$5 = 0
FOR _帧数
    $2 = 1
    $1 += 1
    IF $1 = 31
        $1 = 1
        $2 = 2
    ENDIF
    $5 += 1
    FOR $2
        A 50
        WAIT $4
        LS LEFT
        WAIT $3
        RS LEFT
        WAIT $3
        LEFT DOWN
        WAIT $3
        LS UP
        WAIT $3
        RIGHT DOWN
        WAIT $3
        LS RIGHT
        WAIT $3
        RS RIGHT
        WAIT $3
        LEFT UP
        RIGHT UP
        A 50
        WAIT $4
    NEXT
    PRINT 过了 & $5 & 帧
NEXT
